# Шаг 1: сообщаем git, какие файлы хотим включить в коммит
# "." означает "все файлы в текущей папке"
git add .

# Шаг 2: создаём точку сохранения (commit) с описанием изменений
# -m = message (сообщение коммита)
git commit -m "message(сообщение)"

# Шаг 3: отправляем изменения на удалённый сервер (например GitHub)
# без этого коммиты остаются только на твоём компьютере
git push

#выведет log комитов не локально
git log --oneline --graph

# Проверить состояние проекта:
# есть ли локальные изменения и отстаёт ли ветка от origin/main
git status

# Скачать свежую информацию с GitHub, но не менять файлы проекта
git fetch

# Посмотреть новые коммиты, которых у меня ещё нет
git log --oneline HEAD..origin/main

# Посмотреть, какие файлы изменились на GitHub
git diff --name-only HEAD origin/main

# Посмотреть полный список изменений по строкам
git diff HEAD origin/main

# Подтянуть изменения из GitHub в локальный проект
git pull

# Временно спрятать свои незакоммиченные изменения
git stash

# Вернуть спрятанные изменения обратно
git stash pop

# Подготовить все изменения к коммиту
git add .

# Создать локальный коммит
git commit -m "message"

# Отправить свои коммиты на GitHub
git push

# Показать короткую историю коммитов
git log --oneline

# Показать изменения конкретного коммита
git show HASH
